### Génération Dynamique (Next.js)

```typescript
// src/app/sitemap.ts
import { MetadataRoute } from 'next';
import { villes } from '@/data/villes';
import { formations } from '@/data/formations';

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const baseUrl = 'https://formaroute.fr';
  
  // Pages statiques
  const staticPages = [
    { url: baseUrl, priority: 1.0, changeFrequency: 'weekly' },
    { url: `${baseUrl}/formations`, priority: 0.9 },
    { url: `${baseUrl}/tarifs`, priority: 0.9 },
    { url: `${baseUrl}/contact`, priority: 0.8 },
    { url: `${baseUrl}/faq`, priority: 0.7 },
  ];
  
  // Pages formations
  const formationPages = formations.map((f) => ({
    url: `${baseUrl}/formations/${f.slug}`,
    priority: 0.8,
  }));
  
  // Pages SEO locales
  const villePages = villes.map((v) => ({
    url: `${baseUrl}/auto-ecole-${v.slug}`,
    priority: v.priority === 'haute' ? 0.9 : 0.7,
  }));
  
  return [...staticPages, ...formationPages, ...villePages];
}
```
